git init. //对当前目录初始化
git status //随时查看git信息
git show + commit号（随机的） //查看改动的细节
git log //查看提交的日志信息
git diff 未暂存的文件与之前暂存区域的差别
git diff -cached/-staged 暂存的文件与之前提交区域的差别

git add + 文件名 //本地代码-》待提交列表
rm + 文件名  ---》》》》 git rm + 文件名 // 删除待提交列表中的
git rm --cached +文件名 //希望保留在当前工作目录中
git commit -m + 提示信息 //待提交列表-》本地仓库
git commit -a + 提示信息 //本地代码--》本地仓库
git checkout -- 文件名  //撤销本地修改文件后未提交(还未提交到待提交列表中)的操作  待提交列表-》本地代码
git reset --soft HEAD~1 最上面的开始撤销，本地仓库-》待提交列表（如果没有soft，本地仓库-》本地代码）
vim .gitignore //加入忽略的文件 
git remote add origin + URL //将本地仓库推送到远程仓库中去
git push -u origin master	
git push(上传)/pull  + origin master（下载，首次用clone + URL）

git branch 查看本地分支
git branch -b +分支名 //创建新分支
git checkout + 分支名 // 切换到该分支
git branch -d + 分支名 //删除分支
git merge -d +分支名  //将其他分支合并到master分支上(切换到master上执行的命令)
